def on_starting(server):
    """
    실행: Master 프로세스에서 1번만
    시점: Worker 생성 전
    PID: Master PID
    """

    # ✅ 할 수 있는 것:
    # - 전역 설정 (환경 변수, 시스템 설정)
    # - 로깅 설정
    # - 파일 디스크립터 제한 증가
    # - 설정 파일 읽기

    import resource
    resource.setrlimit(resource.RLIMIT_NOFILE, (65536, 65536))

    # ❌ 하지 말아야 할 것:
    # - 네트워크 연결 생성 (Redis, DB)
    # - Worker별 리소스 할당

2. when_ready (Master, 1번)
def when_ready(server):
    """
    실행: Master 프로세스에서 1번만
    시점: 소켓 바인딩 완료 후, Worker 생성 전
    PID: Master PID
    """

    # ✅ 할 수 있는 것:
    # - 준비 완료 알림 (파일 생성, 시그널 등)
    # - 모니터링 시작
    # - Health check 엔드포인트 활성화

    with open('/tmp/gunicorn-ready', 'w') as f:
        f.write(str(os.getpid()))

    server.log.info(f"Server ready on {server.cfg.bind}")

3. pre_fork (Master, Worker 수만큼)
def pre_fork(server, worker):
    """
    실행: Master 프로세스에서 각 Worker마다
    시점: fork() 직전
    PID: Master PID (항상 같음)
    worker.pid: None (아직 생성 전)
    worker.age: 0, 1, 2, 3... (워커 인덱스)
    """

    # ✅ 할 수 있는 것:
    # - Master의 불필요한 리소스 정리
    # - Worker 생성 전 로깅
    # - 메모리 정리

    import gc
    gc.collect()  # 가비지 컬렉션

    server.log.debug(f"About to fork worker {worker.age}")

    # ❌ 하지 말아야 할 것:
    # - 네트워크 연결 생성
    # - Worker용 리소스 할당

    # ⚠️ 주의:
    # 여기서 생성한 객체는 fork로 Worker가 상속받음
    # → 공유 문제 발생!

4. post_fork (Worker, Worker당 1번)
def post_fork(server, worker):
    """
    실행: 각 Worker 프로세스에서 1번씩
    시점: fork() 직후
    PID: Worker PID (각각 다름)
    worker.pid: Worker PID
    worker.age: 0, 1, 2, 3...
    """

    # ✅ 할 수 있는 것: (Worker별 독립 리소스)
    # - Redis 연결 생성
    # - DB 커넥션 풀 생성
    # - 캐시 초기화
    # - Worker별 로거 설정
    # - 프로세스 이름 변경

    from common.redis_connector import AsyncRedisConnector
    import logging

    # 프로세스 이름
    multiprocessing.current_process().name = f'worker-{worker.age}'

    # Worker별 Redis (독립적)
    logger = logging.getLogger(__name__)
    worker._redis = AsyncRedisConnector(logger, db=0, reload=True)

    # Worker별 DB Pool (독립적)
    worker._db = create_db_pool()

    server.log.info(
        f"Worker initialized: "
        f"PID={worker.pid}, "
        f"age={worker.age}, "
        f"name={multiprocessing.current_process().name}"
    )

    # ❌ 하지 말아야 할 것:
    # - Master의 리소스에 접근
    # - 다른 Worker와 공유 리소스 생성 (파일 락 등)

5. worker_exit (Worker, Worker당 1번)
def worker_exit(server, worker):
    """
    실행: 각 Worker 프로세스에서 종료 시
    시점: Worker 종료 직전
    PID: Worker PID
    """

    # ✅ 할 수 있는 것:
    # - Worker가 생성한 리소스 정리
    # - 연결 종료
    # - 임시 파일 삭제
    # - 최종 로깅

    server.log.info(f"Worker exiting: PID={worker.pid}")

    # Redis 정리
    if hasattr(worker, '_redis'):
        worker._redis.stop_listening()
        del worker._redis

    # DB 정리
    if hasattr(worker, '_db'):
        worker._db.close_all()
        del worker._db

    # 리소스 트래커 정리
    try:
        multiprocessing.resource_tracker._resource_tracker._fd = None
    except:
        pass

6. on_exit (Master, 1번)
def on_exit(server):
    """
    실행: Master 프로세스에서 1번만
    시점: 모든 Worker 종료 후, Master 종료 직전
    PID: Master PID
    """

    # ✅ 할 수 있는 것:
    # - 전역 리소스 정리
    # - 최종 로그
    # - PID 파일 삭제
    # - 정리 신호 전송

    server.log.info("Master process exiting")

    # PID 파일 삭제
    if os.path.exists('/tmp/gunicorn-ready'):
        os.remove('/tmp/gunicorn-ready')

    # 최종 통계
    server.log.info(f"Total requests processed: {server.total_requests}")

## Worker 재시작 시나리오
```
Worker 1이 크래시 (또는 max_requests 도달)
  ↓
Master가 감지
  ↓
worker_exit(server, worker_1) 실행 (Worker 1 프로세스)
  - Redis 정리
  - DB 정리
  ↓
Worker 1 프로세스 종료
  ↓
Master가 새 Worker 생성 시작
  ↓
pre_fork(server, new_worker) 실행 (Master)
  ↓
fork()
  ↓
post_fork(server, new_worker) 실행 (새 Worker)
  - 새 Redis 연결
  - 새 DB Pool
  ↓
새 Worker 준비 완료 (PID: 1005)
